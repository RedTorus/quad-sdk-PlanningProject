106 void loadEigenVectorsFromParams()
107 {
108 // Load the gravity vector
109 g_vec << 0, 0, -g;
110
111 // Load the reachability test points
112 reachability_points_body << 0.5 * robot_l, -0.5 * robot_l, 0.5 * robot_l,
113 -0.5 * robot_l, 0.5 * robot_w, 0.5 * robot_w, -0.5 * robot_w,
114 -0.5 * robot_w, 0, 0, 0, 0;
115 // Load the collision test points
116 collision_points_body << 0.5 * robot_l, -0.5 * robot_l, 0.5 * robot_l,
117 -0.5 * robot_l, 0, 0.5 * robot_w, 0.5 * robot_w, -0.5 * robot_w,
118 -0.5 * robot_w, 0, -0.5 * robot_h, -0.5 * robot_h, -0.5 * robot_h,
119 -0.5 * robot_h, -0.5 * robot_h;
120 }

217 bool operator==(const State rhs) const
218 {
219 // Z velocity is overridden by action
220 return ((pos == rhs.pos) && (vel.head<2>() == rhs.vel.head<2>()));
221 }
222
223 bool operator!=(const State rhs) const
224 {
225 // Z velocity is overridden by action
226 return ((pos != rhs.pos) || (vel.head<2>() != rhs.vel.head<2>()));
227 }
228
229 bool isApprox(const State rhs) const
230 {
231 // Z velocity is overridden by action
232 return ((pos.isApprox(rhs.pos)) &&
233 (vel.head<2>().isApprox(rhs.vel.head<2>())));
234 }
