104 void loadEigenVectorsFromParams()
105 {
106 // Load the gravity vector
107 g_vec << 0, 0, -g;
108
109 // Load the reachability test points
110 reachability_points_body << 0.5 * robot_l, -0.5 * robot_l, 0.5 * robot_l,
111 -0.5 * robot_l, 0.5 * robot_w, 0.5 * robot_w, -0.5 * robot_w,
112 -0.5 * robot_w, 0, 0, 0, 0;
113 // Load the collision test points
114 collision_points_body << 0.5 * robot_l, -0.5 * robot_l, 0.5 * robot_l,
115 -0.5 * robot_l, 0, 0.5 * robot_w, 0.5 * robot_w, -0.5 * robot_w,
116 -0.5 * robot_w, 0, -0.5 * robot_h, -0.5 * robot_h, -0.5 * robot_h,
117 -0.5 * robot_h, -0.5 * robot_h;
118 }

215 bool operator==(const State rhs) const
216 {
217 // Z velocity is overridden by action
218 return ((pos == rhs.pos) && (vel.head<2>() == rhs.vel.head<2>()));
219 }
220
221 bool operator!=(const State rhs) const
222 {
223 // Z velocity is overridden by action
224 return ((pos != rhs.pos) || (vel.head<2>() != rhs.vel.head<2>()));
225 }
226
227 bool isApprox(const State rhs) const
228 {
229 // Z velocity is overridden by action
230 return ((pos.isApprox(rhs.pos)) &&
231 (vel.head<2>().isApprox(rhs.vel.head<2>())));
232 }
